class CategoryScreen extends StatelessWidget {

  final ScrollController _scrollController = ScrollController();

  CategoryScreen({
    super.key,

  void _scrollToTop() {
    _scrollController.animateTo(
      0, // ‡∏ï‡∏≥‡πÅ‡∏´‡∏ô‡πà‡∏á‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô (‡∏î‡πâ‡∏≤‡∏ô‡∏ö‡∏ô‡∏™‡∏∏‡∏î)

  @override

        title: Text(title),

        onPressed: _scrollToTop, // ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡πÄ‡∏•‡∏∑‡πà‡∏≠‡∏ô‡∏Å‡∏•‡∏±‡∏ö‡∏î‡πâ‡∏≤‡∏ô‡∏ö‡∏ô

          borderRadius: BorderRadius.circular(20), // ‡∏°‡∏∏‡∏°‡∏°‡∏ô
